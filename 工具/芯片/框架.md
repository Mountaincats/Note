以下是一个结合了NVBOARD交互式图形界面、波形记录和完整时钟控制的Verilator仿真框架：

```cpp
// 综合仿真框架：NVBOARD交互 + 波形记录 + 完整时钟控制
#include <nvboard.h>           // NVBOARD图形界面库
#include <verilated.h>         // Verilator仿真内核
#include <verilated_fst_c.h>   // FST格式波形记录
#include "Vtop.h"              // 你的Verilog顶层模块生成的C++模型

// 全局仿真对象声明
static Vtop dut;               // 设计实例
static VerilatedContext* contextp = nullptr;  // 仿真上下文
static VerilatedFstC* tfp = nullptr;         // 波形文件对象

// NVBOARD引脚绑定函数（需根据实际设计定义）
void nvboard_bind_all_pins(Vtop* top) {
    // 示例：绑定开关、LED、按键等
    // 请根据你的设计修改
    nvboard_bind_pin(&top->clk, 1, BTN0);   // 时钟连接到按钮0
    nvboard_bind_pin(&top->rst, 1, BTN1);   // 复位连接到按钮1
    nvboard_bind_pin(&top->sw, 8, SW7, SW6, SW5, SW4, SW3, SW2, SW1, SW0);
    nvboard_bind_pin(&top->led, 8, LD7, LD6, LD5, LD4, LD3, LD2, LD1, LD0);
    nvboard_bind_pin(&top->seg, 7, SEG0A, SEG0B, SEG0C, SEG0D, SEG0E, SEG0F, SEG0G);
    nvboard_bind_pin(&top->an, 8, AN0, AN1, AN2, AN3, AN4, AN5, AN6, AN7);
}

// 单时钟周期驱动
static void single_cycle() {
    dut.clk = 0;
    dut.eval();
    if (tfp) tfp->dump(contextp->time());  // 记录时钟下降沿
    
    contextp->timeInc(1);  // 时间推进
    
    dut.clk = 1;
    dut.eval();
    if (tfp) tfp->dump(contextp->time());  // 记录时钟上升沿
    
    contextp->timeInc(1);  // 时间推进
}

// 复位函数
static void reset(int n) {
    dut.rst = 1;
    for (int i = 0; i < n; i++) {
        single_cycle();
    }
    dut.rst = 0;
}

int main(int argc, char** argv) {
    // 1. 初始化Verilator仿真上下文
    contextp = new VerilatedContext;
    contextp->commandArgs(argc, argv);
    
    // 2. 初始化设计实例（注意：这里使用全局变量dut，已在定义时用contextp初始化）
    // 如果dut需要contextp初始化，可以在定义时使用Vtop dut{contextp.get()}
    
    // 3. 初始化波形记录
    bool enable_trace = false;
    for (int i = 1; i < argc; i++) {
        if (strcmp(argv[i], "--trace") == 0) {
            enable_trace = true;
            break;
        }
    }
    
    if (enable_trace) {
        Verilated::traceEverOn(true);
        tfp = new VerilatedFstC;
        dut.trace(tfp, 99);  // 跟踪99层层次结构
        tfp->open("waveform.fst");
        printf("Waveform tracing enabled. Output: waveform.fst\n");
    }
    
    // 4. 初始化NVBOARD
    nvboard_bind_all_pins(&dut);
    nvboard_init();
    printf("NVBOARD initialized. Press Ctrl+C to exit.\n");
    
    // 5. 执行复位序列
    reset(10);
    printf("Reset completed.\n");
    
    // 6. 主仿真循环
    vluint64_t max_sim_time = 1000000;  // 最大仿真时间（时钟周期数）
    bool user_exit = false;
    
    while (!contextp->gotFinish() &&         // Verilog中没有调用$finish
           contextp->time() < max_sim_time && // 未达到最大仿真时间
           !user_exit) {                     // 用户没有退出
        
        // 6.1 更新NVBOARD界面（读取输入/刷新显示）
        nvboard_update();
        
        // 6.2 驱动一个时钟周期
        single_cycle();
        
        // 6.3 可选的调试信息输出
        if (contextp->time() % 1000 == 0) {
            // 每1000个时间单位输出一次状态
            printf("Simulation time: %lu\n", contextp->time());
            
            // 示例：输出一些关键信号状态
            // printf("LED state: 0x%02X\n", dut.led);
        }
        
        // 6.4 检查是否收到退出信号（如窗口关闭）
        // NVBOARD内部会处理SDL事件，包括退出
        // 这里可以添加自定义的退出条件检查
    }
    
    // 7. 仿真结束，清理资源
    printf("Simulation completed.\n");
    printf("Total simulation time: %lu cycles\n", contextp->time());
    
    nvboard_quit();  // 关闭NVBOARD窗口
    
    if (tfp) {
        tfp->close();
        delete tfp;
        printf("Waveform saved to waveform.fst\n");
    }
    
    delete contextp;
    
    return 0;
}
```

## 编译脚本示例 (Makefile)

```makefile
# 编译设置
VERILATOR = verilator
NVBOARD_DIR = /usr/local/share/nvboard
VERILATOR_DIR = /usr/local/share/verilator

# 设计文件
VERILOG_SOURCES = top.v
TOP_MODULE = top

# 生成的文件
VERILATED_DIR = obj_dir
VERILATED_HEADER = $(VERILATED_DIR)/V$(TOP_MODULE).h
VERILATED_SO = $(VERILATED_DIR)/V$(TOP_MODULE)__ALL.a

# 编译选项
VERILATOR_FLAGS = -Wall --cc --exe --build
INCLUDES = -I$(NVBOARD_DIR)/include -I$(VERILATOR_DIR)/include
LIBS = -L$(NVBOARD_DIR)/lib -lnvboard \
       -L$(VERILATOR_DIR)/lib -lverilated -lverilated_fst

# 默认目标
all: sim_top

# 生成Verilator模型
$(VERILATED_SO): $(VERILOG_SOURCES)
	$(VERILATOR) $(VERILATOR_FLAGS) $(VERILOG_SOURCES) --top-module $(TOP_MODULE)

# 编译主程序
main.o: main.cpp $(VERILATED_HEADER)
	g++ -std=c++11 $(INCLUDES) -c main.cpp -o main.o

# 链接生成可执行文件
sim_top: $(VERILATED_SO) main.o
	g++ -std=c++11 $(INCLUDES) main.o $(VERILATED_SO) $(LIBS) -o sim_top

# 运行仿真
run: sim_top
	./sim_top --trace  # 启用波形记录

# 查看波形
wave:
	gtkwave waveform.fst &

# 清理
clean:
	rm -rf $(VERILATED_DIR) *.o sim_top waveform.fst

.PHONY: all run wave clean
```
